\begin{frame}[fragile]

  \begin{lstlisting}

\begin{frame}[fragile]
  \frametitle{Modules and function, oh my!}
  \begin{lstlisting}
    defmodule Geometry do
      def rectangle_area(a, b) do
        a * b
      end
    end
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Composing functions}
  \begin{lstlisting}
    def process_xml(model, xml) do
      model
      |> update(xml)
      |> process_changes
      |> persist
    end
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Function arity}
  \begin{lstlisting}
    defmodule Rectangle do
      def area(a), do: area(a, a)
      def area(a, b), do: a * b
    end
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Typespec}
  \begin{lstlisting}
    % read up on dializer and custom types
    defmodule Circle do
      @pi 3.14159

      @spec area(number) :: number
      def area(r), do: r*r*@pi
        
      @spec circumference(number) :: number
      def circumference(r), do: 2*r*@pi
    end
  \end{lstlisting}
\end{frame}
